93)Display those employees whose salary greter than his manager salary. 

  SELECT e1.first_name AS employee_name, e1.salary AS employee_salary, e2.last_name AS manager_name, e2.salary AS manager_salary
  FROM employees e1
  JOIN employees e2 ON e1.manager_id = e2.employee_id WHERE e1.salary > e2.salary;
